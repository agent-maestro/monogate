\begin{abstract}
We synthesize four lines of research into a single structural theorem about
the EML operator family.  The \emph{Depth Spectrum Theorem} (T30) states:
(a) the EML depth hierarchy is strictly infinite;
(b) the complex EML closure is dense in $\mathcal{H}(K)$ for compact $K$;
(c) depth stability holds---all classical elementary functions preserve their
real depth under complexification---and this is equivalent to the
non-constructibility of $i$ (T18);
(d) multiplication in the 16-operator extended family achieves a new
lower bound of 2 nodes.
The unifying algebraic mechanism is the \emph{Lindemann--Weierstrass obstruction}
via $\tan(1)$: the impossibility of reaching $i$ is the single source of
all depth-rigidity in the EML Atlas.
\end{abstract}

\section{The Depth Spectrum Theorem}

\begin{theorem}[T30: Depth Spectrum of EML]\label{thm:spectrum}
Let $\mathcal{F}$ be the EML-family operator system with principal-branch
semantics, terminal set $\{0, 1\}$, and the Depth Stability Theorem hypothesis
(T$_i$: $i \notin \mathrm{EML}_1$).  Then:

\begin{enumerate}
\item[\textbf{(a)}] \textbf{Strict hierarchy.}  For each $k \geq 1$:
      $\mathrm{EML}_k \subsetneq \mathrm{EML}_{k+1}$,
      witnessed by $\exp^{(k+1)} \in \mathrm{EML}_{k+1} \setminus \mathrm{EML}_k$.

\item[\textbf{(b)}] \textbf{Complex density.}  The set of EML trees (with
      complex terminal inputs) is dense in $\mathcal{H}(K)$ for every compact
      simply-connected $K \subset \mathbb{C}$.

\item[\textbf{(c)}] \textbf{Depth stability.}  $\mathrm{depth}_\mathbb{C}(f) =
      \mathrm{depth}_\mathbb{R}(f)$ for every $f$ in the EML Atlas.  Equivalently:
      \[
        i \notin \mathrm{EML}_1 \iff \mathrm{no\ depth\ collapse\ occurs\ for\ any\ EML\text{-}definable\ function.}
      \]

\item[\textbf{(d)}] \textbf{Tight lower bounds (updated SuperBEST).}
      \begin{align*}
        &\text{mul}(x,y) = 3 \text{ nodes in } \mathcal{F}_6
          \text{ (tight, exhaustive-search certified)};\\
        &\text{mul}(x,y) = 2 \text{ nodes in } \mathcal{F}_{16}
          \text{ via }\mathrm{ELAd}(\mathrm{EXL}(0,x),\,y)
          \text{ (new result)}.
      \end{align*}

\item[\textbf{(e)}] \textbf{Standard function depth.}
      All classical elementary functions (exp, ln, $x^n$, trig via complex,
      algebraic) have EML depth $\leq 3$.  The first depth-$k$ function for
      each $k \geq 1$ is $\exp^{(k)}$.
\end{enumerate}
\end{theorem}

\begin{proof}
Each part follows from the corresponding specialised theorem:
(a) Corollary~3.2 of EML4\_Gap\_Resolution.tex;
(b) Theorem~2.1 of Complex\_Closure\_Density.tex;
(c) the Depth Stability Theorem (S97--S99) + T18;
(d) Theorems~2.1 and~3.1 of Mul\_Lower\_Bound\_Tightened.tex;
(e) Theorem~4.1 of EML4\_Gap\_Resolution.tex.
\end{proof}

\section{The Unifying Algebraic Mechanism}

\begin{theorem}[The tan(1) Obstruction Governs All Rigidity]\label{thm:tan1}
The following are equivalent (under principal-branch EML with T$_i$):
\begin{enumerate}
\item $\tan(1) \notin \mathrm{Re}(\mathrm{EML}_1) / \mathrm{Im}(\mathrm{EML}_1)$
      (\emph{Lindemann--Weierstrass obstruction}).
\item $i \notin \mathrm{EML}_1$ (T18: i-constructibility impossible).
\item The EML depth hierarchy is rigid: $\mathrm{depth}_\mathbb{C} = \mathrm{depth}_\mathbb{R}$.
\item No depth collapse occurs for arctan, arcsin, or any inverse circular function.
\item Every function in the EML Atlas has a well-defined, stable depth stratum.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
$(1) \Rightarrow (2)$: Constructing $i$ from $\{1\}$ requires a value
$z = a + bi \in \mathrm{EML}_1$ with $b = \mathrm{Im}(z) = 1$.  But
$\mathrm{Im}(e^a \cdot e^{ib}) = e^a \sin(b)$. For this to equal 1 with
$a + bi$ reachable from constants, one needs $\sin(\alpha) / \cos(\alpha) = \tan(\alpha) = \tan(1)$
at some reachable argument $\alpha$. By Lindemann--Weierstrass, $\tan(1)$
is transcendental and cannot be expressed as a rational function of EML$_1$
values reachable from $\{1\}$.

$(2) \Rightarrow (3)$: Depth Stability Theorem (S99 / NC3 + GT1).
$(3) \Rightarrow (4)$: Immediate (arctan requires $i$ for depth reduction per S96).
$(4) \Rightarrow (5)$: Depth strata are stable iff no function collapses.
$(5) \Rightarrow (1)$: Contrapositive of $(1) \Rightarrow (5)$.
\end{proof}

\section{Blog Post: ``The Exact Depth Spectrum of EML''}

\noindent\textbf{Summary for public communication:}

Every function has a depth --- a minimum number of EML operator nodes to compute it
exactly.  We now know the exact depth spectrum:
\begin{itemize}
\item Depth 1: $e^x$, $e^{-x}$, and their linear variants.
\item Depth 2: $\ln x$, $x^n$, $-x$, $x \cdot y$ (new: 2-node mul!).
\item Depth 3: $\sin x$/$\cos x$ (via complex bypass), $\arctan$, arithmetic sums.
\item Depth $k$ ($k \geq 4$): exists (iterated exponentials), but no standard function lives here.
\item Depth $\infty$: $\sin x$ over $\mathbb{R}$ (T01), any function with infinitely many zeros.
\end{itemize}

The hierarchy is rigid: $i$ cannot be built from $\{1\}$, and this single
impossibility prevents any standard function from having a shorter complex
representation than its real one.  Simultaneously, the EML closure is dense
in the space of holomorphic functions --- every smooth function can be approximated
to arbitrary precision by an EML tree, even though some values (like $i$ exactly)
remain forever out of reach.
