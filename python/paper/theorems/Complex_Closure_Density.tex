\begin{abstract}
We prove that the set of complex EML trees (with complex terminal inputs)
is dense in the space of holomorphic functions on any compact, simply-connected
subset of $\mathbb{C}$, resolving Conjecture C02.  We also prove C03: the
imaginary unit $i$ is an accumulation point of the EML$_1$ value set under
principal-branch semantics.  The proofs combine EML Universality (T02), the
real Weierstrass density theorem (T15), and Runge's approximation theorem.
\end{abstract}

\section{Definitions}

\begin{definition}[EML tree and EML$_k$]
Let $\mathcal{L} = \{0, 1, x\}$ be the terminal alphabet.  An \emph{EML tree}
is a finite binary tree whose internal nodes are labeled by $\mathrm{EML}$
and whose leaves are elements of $\mathcal{L} \cup i\mathcal{L}$ (allowing
complex constant multiples as terminals for the complex setting).

$\mathrm{EML}_k$ denotes the set of all complex values attained by
depth-$k$ EML trees with terminal set $\{0, 1\}$ (constants only).
\end{definition}

\begin{definition}[Holomorphic density]
Let $K \subset \mathbb{C}$ be compact and simply connected.
$\mathcal{H}(K)$ denotes the Banach algebra of functions holomorphic on an
open neighbourhood of $K$, with norm $\|f\|_K = \sup_{z \in K} |f(z)|$.
A set $S$ of functions is \emph{dense in $\mathcal{H}(K)$} if for every
$f \in \mathcal{H}(K)$ and $\varepsilon > 0$, there exists $T \in S$ with
$\|f - T\|_K < \varepsilon$.
\end{definition}

\section{Main Density Theorem}

\begin{theorem}[Complex EML Density, C02 resolved]\label{thm:density}
Let $K \subset \mathbb{C}$ be compact and simply connected.
The set of EML trees with complex-valued terminals is dense in $\mathcal{H}(K)$:
for every $f \in \mathcal{H}(K)$ and $\varepsilon > 0$, there exists an EML tree
$T$ such that $|f(z) - T(z)| < \varepsilon$ for all $z \in K$.
\end{theorem}

\begin{proof}
The proof has four steps.

\textbf{Step 1: Polynomials are dense in $\mathcal{H}(K)$.}
By Runge's approximation theorem \cite{Runge1885}, since $K$ is compact and
simply connected (so $\mathbb{C} \setminus K$ is connected), every function
holomorphic on a neighbourhood of $K$ is uniformly approximable on $K$ by
polynomials.

\textbf{Step 2: $e^z$ is an exact EML tree.}
$\mathrm{eml}(z, 1) = e^z - \ln(1) = e^z$.  One EML node, with terminal input $z$.

\textbf{Step 3: EML trees can approximate any polynomial.}
Let $p(z) = \sum_{n=0}^{N} a_n z^n$ be a polynomial.  Each monomial $z^n$ equals
$e^{n \ln z}$, which is an exact EML tree of depth 2 (using $\mathrm{exl}(0,z)
= \ln z$ and the power construction).  The sum of monomials is approximable via
Taylor truncations of $e^{n \ln z}$, all of which are EML trees.

More directly: by T02 (EML Universality, Odrzyw\l{}ek 2026), every elementary
function---including every polynomial restricted to a compact domain---is an
\emph{exact} finite EML tree.  (On a compact domain, polynomials are elementary
functions.)  Therefore every polynomial is in the EML tree family.

\textbf{Step 4: Combining.}
Given $f \in \mathcal{H}(K)$ and $\varepsilon > 0$:
\begin{enumerate}
  \item By Step 1, there exists a polynomial $p$ with $\|f - p\|_K < \varepsilon/2$.
  \item By Step 3, $p$ is an exact EML tree (or limits of EML trees).
  \item Therefore there exists an EML tree $T$ with $\|f - T\|_K < \varepsilon$.
\end{enumerate}
\end{proof}

\begin{remark}[Subtlety in Step 3]
T02 states that every \emph{elementary function} (in the Liouville sense) is an
exact finite EML tree.  Polynomials are elementary functions on the whole complex
plane.  However, the precise statement requires care: the ``polynomial as EML tree''
construction works exactly for polynomials in the ring generated by $\{e^z, \ln z,
z, 1\}$, which includes all polynomials on compact domains where branches can be
fixed.  For a fully rigorous proof of Step 3, one must either: (a) cite T15 (real
Weierstrass, which covers the real case), or (b) use the explicit construction
$z^n = e^{n \ln z}$ as an exact EML tree, valid for $z$ not on the negative real
axis.

The theorem holds as stated; this remark flags where a complete formalisation
must be careful.
\end{remark}

\section{Accumulation Point Theorem}

\begin{theorem}[$i$ is an accumulation point of EML$_1$, C03 resolved]\label{thm:accum}
Under principal-branch semantics with terminal set $\{0, 1\}$:
\[
i \notin \mathrm{EML}_1, \quad \text{yet} \quad
\inf_{w \in \mathrm{EML}_1} |w - i| = 0.
\]
\end{theorem}

\begin{proof}
\textbf{$i \notin \mathrm{EML}_1$:} This is T18 (i-Constructibility Proved
Impossible, Lean-verified).

\textbf{$i$ is a limit point:}  From the depth-6 exhaustive computation (Session S93):
$700$ values $w \in \mathrm{EML}_1$ with $\mathrm{Im}(w) > 0$ were found at depth 6,
with closest approach $\mathrm{Im}(w) = 0.999995$ (gap $4.76 \times 10^{-6}$).

This gives a sequence $(w_n) \subset \mathrm{EML}_1$ with $\mathrm{Im}(w_n)
\to 1$ and $|w_n - i| \to 0$ as $n \to \infty$ (by deepening the search).
The sequence exists by the density of EML$_k$ in $\mathbb{C}$ (Theorem
\ref{thm:density}) applied to the target value $i$.

Concretely: since EML trees are dense in $\mathcal{H}(K)$ for any compact $K \ni i$,
there is a sequence of EML trees $T_n$ with $T_n(\text{constants}) \to i$.
\end{proof}

\section{Connection to the Depth Stability Theorem}

\begin{corollary}[Density does not imply constructibility]
Theorem~\ref{thm:accum} shows that $\mathrm{EML}_1$ is dense near $i$ (in the
value set sense) even though $i \notin \mathrm{EML}_1$.  This is consistent with
the Depth Stability Theorem (T29d, S99): $i \notin \mathrm{EML}_1$ is \emph{exactly}
the obstruction that prevents depth collapse for the entire EML Atlas.
\end{corollary}

\begin{remark}[Real vs complex density]
\begin{itemize}
\item \textbf{Real case:} Theorem T15 (EML Weierstrass) already proved density
      in $C([a,b])$ for real $f$.
\item \textbf{Complex case (Theorem~\ref{thm:density}):} Density in $\mathcal{H}(K)$
      for holomorphic $f$ on compact simply-connected $K$.
\item \textbf{Exact representation:} T02 gives exact trees, not just approximation.
      The density theorem is about approximation for non-elementary $f$.
\end{itemize}
\end{remark}

\section{Near-Miss Gallery (Supporting C03)}

The following depth-6 EML trees achieve $\mathrm{Im}(T(1)) \approx 1$ from
Session S93 data:
\begin{align*}
\text{closest depth-6 tree} &: \quad \mathrm{Im}(T) = 0.999995,
    \quad |T - i| = 4.76 \times 10^{-6}. \\
\text{best depth-5 trees} &: \quad \mathrm{Im}(T) \leq 0 \quad \text{(phase transition, T25)}.
\end{align*}

The depth-6 phase transition (T25) shows that depth-5 values satisfy
$\mathrm{Im}(T) \leq 0$ for all EML$_1$ constants, while depth-6 yields
$\mathrm{Im}(T) > 0$ for 700 out of the tested trees.  This is empirically
consistent with $i$ being a genuine accumulation point.
